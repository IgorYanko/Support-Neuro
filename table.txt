CREATE TYPE type_enum AS ENUM ('Venda', 'Assistência Técnica', 'Orçamento', 'Compra', 'Venda Consignada', 'Venda Representação', 'Bonificação/Remessa', 'Ordem de Serviço', 'Transferência');
CREATE TYPE status_enum AS ENUM ('Em aberto', 'Em orçamento', 'Aprovado', 'Faturado', 'Orçamento Recusado', 'Cancelado');

CREATE TABLE serviceOrders (
	id SERIAL PRIMARY KEY,
	numOS VARCHAR(10) NOT NULL,
	nome VARCHAR(100),
	razao VARCHAR(100),
	arrivalDate VARCHAR(30) NOT NULL,
	osType type_enum NOT NULL,
	osStatus status_type NOT NULL,
	observations VARCHAR(250),
	warranty BOOLEAN NOT NULL DEFAULT FALSE
);
